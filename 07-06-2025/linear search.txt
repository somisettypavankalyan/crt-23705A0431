class Codechef {
    public static void main(String[] args)  {
        int[] num = {1, 5, 9, 6, 3, 6};
        int target = 6;
        int index = -1;

        
        for (int i = 0; i < num.length; i++) {
            if (target == num[i]) {
                index = i; 
            }
        }

        System.out.println(index); 
    }
}
